 31     public function exportEntry($entryId, $campaignId, $return = false)
 32     {
 33         // Define our variables
 34         $listIds = array ();
 35         $listProviders = array ();
 36 
 37         // Get our campaign info
 38         if ( ! $campaign = craft()->sproutEmail_campaign->getCampaignById($campaignId) )
 39         {
 40             SproutEmailPlugin::log("Campaign not found");
 41 
 42             if ( $return )
 43             {
 44                 return false;
 45             }
 46

 48             die( 'Campaign not found' );
 49         }
 50 
 51         // Get our recipient list info
 52         if($campaign["emailProvider"] != 'CopyPaste')
 53         {
 54             if ( ! $recipientLists = craft()->sproutEmail_campaign->getCampaignRecipientLists( $campaign ['id'] ) )
 55             {
 56                 SproutEmailPlugin::log("Recipient lists not found");
 57 
 58                 if ( $return )
 59                 {
 60                     return false;
 61                 }
 62 
 63                 // @TODO - update use of die
 64                 die( 'Recipient lists not found' );
 65             }
 66         }
 67         else
 68         {
 69             SproutEmailPlugin::log("Exporting Copy/Paste Email");
 70 
 71             // We can't check for a recipients list on CopyPaste since one doesn't exist
 72             craft()->sproutEmail_copyPaste->exportEntry( $campaign );
 73 
 74             // @TODO - update use of die
 75             die();
 76         }
 77 
 78         // Check to see if we have entry level settings and update
 79         // before shuffling off to the individual service connectors.
 80 
 81         $entry = craft()->entries->getEntryById($entryId);
 82         $entryFields = $entry->getFieldLayout()->getFields();
 83 
 84         // Assume we have no override, and update overrideHandle if we do
 85         $overrideHandle = "";
 86 
 87         foreach ($entryFields as $field)
 88         {
 89             // If we have an Email Campaign Field, grab the handle of the first one that matches
 90             if ($field->getField()->type == 'SproutEmail_EmailCampaign')
 91             {
 92                 SproutEmailPlugin::log('We have a Campaign Override field');
 93                 SproutEmailPlugin::log('Override Field Handle: ' . $field->getField()->handle);
 94 
 95                 $overrideHandle = $field->getField()->handle;
 96                 continue;
 97             }
 98         }
 99 
100         // If the entry has an override handle assigned to it
101         if ($overrideHandle != "")
102         {
103             // Grab our Email Campaign override settings
104             $entryOverrideSettings = $entry->{$overrideHandle};
105             $entryOverrideSettings = json_decode($entryOverrideSettings,TRUE);
106 
107             SproutEmailPlugin::log('Our override settings: ' . $entry->{$overrideHandle});
108 
109             // Merge the entry level settings with our campaign
110             $emailProviderRecipientListId = '';
111 
112             if( isset($entryOverrideSettings) )
113             {
114                 foreach($entryOverrideSettings as $key => $value)
115                 {
116                     // Override our campaign settings
117                     $campaign[$key] = $value;
118 
119                     if($key == 'emailProviderRecipientListId')
120                     {
121                         // Make sure our $emailProviderRecipientListId is an array
122                         // @TODO - clarify what this value is for.  Is it only for Mailgun?
123                         $emailProviderRecipientListId = (array) $value;
124                         $campaign[$key] = $emailProviderRecipientListId;
125                     }
126 
127                     SproutEmailPlugin::log('Entry override ' . $key . ': ' . $value);
128 
129                 }
130             }
131 
132             // @TODO - need to revisit this behavior big time!
133             if($emailProviderRecipientListId != '')
134             {
135                 foreach ($emailProviderRecipientListId as $key => $value)
136                 {
137                     $recipientLists[0]->emailProviderRecipientListId = $entryOverrideSettings['emailProviderRecipientListId']['list'];
138                     $recipientLists[0]->emailProvider = $campaign['emailProvider'];
139                     $recipientLists[0]->type = null;
140 
141                     // $recipientListOverrides[$key] = $recipientListOverride;
142                 }
143             }
144         }
145 
146         // Create the recipient list variables
147         foreach ( $recipientLists as $list )
148         {
149             $listProviders [] = $list->emailProvider;
150             $listIds [$list->emailProvider] [] = $list ['emailProviderRecipientListId'];
151         }
152 
153         foreach ( $listProviders as $provider )
154         {
155             $provider_service = 'sproutEmail_' . lcfirst( $provider );
156             craft()->{$provider_service}->exportEntry( $campaign, $listIds [$provider], $return );
157         }
158 
159         if ( $return )
160         {
161             return true;
162         }
163 
164         // @TODO - update use of die
165         die();
